What's a model in your Angular app?


module.controller('HomeCtrl', function($scope) {
  $scope.person = {
    name: 'Gonto',
    company: 'Auth0'
  };

  $scope.changeName = function(toName) {
    if (toName === 'Martin') {
      alert("No Martins are allowed");
    } else {
      $scope.person = toName;
    }
  };
});

class Person {
  name: string;
  company: number;
}

@Component({
  selector: 'users'
})
@View({
  directives: [FORM_DIRECTIVES],
  templateUrl: 'userTemplate.html'
})
class UsersAppComponent {
  
  person = new Person();

  changeName(toName) {
    if (toName === 'Martin') {
      alert("No Martins are allowed");
    } else {
      person.name = toName;
    }
  }
}

Now can you see why this is not good?

It doesn't mimic the "real world"

Doesn't follow DRY

Behaviour is spread throughout the app

Validations are in the view, not in the model

Can't share models with the backend
